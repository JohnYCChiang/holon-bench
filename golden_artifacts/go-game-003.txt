package room

type Player struct {
	ID        string
	JoinedAt  int
	Connected bool
}

type Room struct {
	ID      string
	HostID  string
	Closed  bool
	Players []Player
}

func New(id string, players []Player) *Room {
	r := &Room{ID: id, Players: append([]Player(nil), players...)}
	r.transferHost()
	return r
}

func (r *Room) Leave(playerID string) {
	remaining := r.Players[:0]
	for _, p := range r.Players {
		if p.ID != playerID {
			remaining = append(remaining, p)
		}
	}
	r.Players = remaining
	r.HostID = ""
	r.Closed = len(r.Players) == 0
	if !r.Closed {
		r.transferHost()
	}
}

func (r *Room) transferHost() {
	var candidate *Player
	for i := range r.Players {
		player := &r.Players[i]
		if !player.Connected {
			continue
		}
		if candidate == nil || player.JoinedAt < candidate.JoinedAt {
			candidate = player
		}
	}
	if candidate != nil {
		r.HostID = candidate.ID
	}
}
