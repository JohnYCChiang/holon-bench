#[derive(Clone, Copy, Debug, PartialEq)]
pub struct Vec2 {
    pub x: f32,
    pub y: f32,
}

#[derive(Clone, Copy, Debug, PartialEq)]
pub struct Bounds {
    pub min: Vec2,
    pub max: Vec2,
}

pub fn follow_position(target: Vec2, offset: Vec2, bounds: Bounds) -> Vec2 {
    let desired = Vec2 { x: target.x + offset.x, y: target.y + offset.y };
    Vec2 {
        x: desired.x.clamp(bounds.min.x, bounds.max.x),
        y: desired.y.clamp(bounds.min.y, bounds.max.y),
    }
}
