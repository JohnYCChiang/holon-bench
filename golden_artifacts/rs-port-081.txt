pub fn byte_offset_to_line_col(text: &str, offset: usize) -> (usize, usize) {
    let clamped = offset.min(text.len());
    let mut line = 1;
    let mut col = 1;
    let mut byte_pos = 0;
    for ch in text.chars() {
        if byte_pos >= clamped {
            break;
        }
        byte_pos += ch.len_utf8();
        if ch == '\n' {
            line += 1;
            col = 1;
        } else {
            col += 1;
        }
    }
    (line, col)
}
