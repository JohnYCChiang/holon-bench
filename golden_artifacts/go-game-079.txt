package replay

import (
	"crypto/sha256"
	"fmt"
	"sort"
	"strings"
)

type Event struct {
	Tick   int
	Player string
	Action string
}

func Checksum(events []Event) string {
	sorted := append([]Event(nil), events...)
	sort.Slice(sorted, func(i, j int) bool {
		if sorted[i].Tick != sorted[j].Tick {
			return sorted[i].Tick < sorted[j].Tick
		}
		if sorted[i].Player != sorted[j].Player {
			return sorted[i].Player < sorted[j].Player
		}
		return sorted[i].Action < sorted[j].Action
	})
	parts := make([]string, 0, len(sorted))
	for _, event := range sorted {
		parts = append(parts, fmt.Sprintf("%d:%s:%s", event.Tick, event.Player, event.Action))
	}
	sum := sha256.Sum256([]byte(strings.Join(parts, "|")))
	return fmt.Sprintf("%x", sum)
}
